Combined example - sonar plus red cup detection
from lesson_header import *
import time

myRobot = bot(base_speed=300)

while True:
    distance_cm = myRobot.sonar.distance_cm(filtered=True)

    if distance_cm == 0:
        myRobot.move.forward(0.1)
    elif distance_cm > 10:
        myRobot.move.forward(0.08)
    else:
        myRobot.stop()
        myRobot.camera.center_all()
        myRobot.vision.calibrate_color("red")
        myRobot.vision.show_color("red")
        red_cup_index = myRobot.vision.which_object("red")
        myRobot.voice.say(f"The red cup is cup {red_cup_index}.", block=True)
        break

    time.sleep(0.1)

Combined example - gesture controlled response
from lesson_header import *

myRobot = bot(base_speed=300)
result = myRobot.vision.recognize_hands()
if result["hands"]:
    gesture = result["hands"][0]["gesture"]
    if gesture == "thumbs_up":
        myRobot.voice.say("Starting now", block=True)
        myRobot.move.forward(0.3)
    elif gesture == "peace":
        myRobot.eyes.set_both(180, 0, 255)
    else:
        myRobot.voice.say(f"I saw {gesture}", block=True)
